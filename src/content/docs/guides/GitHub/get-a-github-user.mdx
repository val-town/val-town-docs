/** @jsxImportSource https://esm.sh/react@18.2.0 */
import { createRoot } from "https://esm.sh/react-dom@18.2.0/client";
import React, { useState } from "https://esm.sh/react@18.2.0";

const NO_PHRASES = [
  "No ðŸ’”",
  "Pretty please? ðŸ¥º",
  "But we'd be so cute together! ðŸ’•",
  "One more chance, pookie?",
  "Don't break my heart :(",
  "What about a maybe?",
  "Please don't do this to me, I'm fragile",
];

function App() {
  const [noClicks, setNoClicks] = useState(0);
  const [isValentine, setIsValentine] = useState(false);
  const [hearts, setHearts] = useState([]);
  const yesButtonSize = noClicks * 20 + 16;

  const firstImg = "https://media.tenor.com/VIChDQ6ejRQAAAAj/jumping-bear-hearts-no-png.gif";
  const secondImg = "https://tenor.com/bGIOY.gif";

  const handleNo = () => {
    setNoClicks((prev) => prev + 1);
  };

  const handleYes = () => {
    setIsValentine(true);

    // Generate falling hearts
    const interval = setInterval(() => {
      const newHeart = {
        id: Math.random(),
        left: Math.random() * 100,
      };
      setHearts((prev) => [...prev, newHeart]);
    }, 300);

    setTimeout(() => clearInterval(interval), 5000);
  };

  return (
    <div>
      {/* Background Music */}
      <audio autoPlay loop>
        <source src="https://dl.sndup.net/jv2k/Can%20I%20Have%20This%20Dance.mp3" type="audio/mpeg" />
      </audio>

      <div
      style={{
        display: "flex",
        flexDirection: "column",
        alignItems: "center",
        justifyContent: "center",
        height: "100vh",
        fontFamily: "Arial, sans-serif",
        textAlign: "center",
        background: "linear-gradient(135deg, #ffe6f0, #fff0f5)",
      }}
    >
      {!isValentine ? (
        <>
          <img src={firstImg} style={{ width: "200px" }} />
          <h1 style={{ color: "#ff4d88" }}>Will you be my Valentine? ðŸ’˜</h1>
          <div>
            <button
              onClick={handleYes}
              style={{
                fontSize: `${yesButtonSize}px`,
                margin: "10px",
                padding: "10px 20px",
                backgroundColor: "#28a745",
                color: "white",
                border: "none",
                borderRadius: "8px",
                cursor: "pointer",
                transition: "all 0.3s ease",
              }}
            >
              Yes ðŸ’–
            </button>
            <button
              onClick={handleNo}
              style={{
                fontSize: "16px",
                margin: "10px",
                padding: "10px 20px",
                backgroundColor: "#dc3545",
                color: "white",
                border: "none",
                borderRadius: "8px",
                cursor: "pointer",
              }}
            >
              {noClicks === 0
                ? "No"
                : NO_PHRASES[Math.min(noClicks - 1, NO_PHRASES.length - 1)]}
            </button>
          </div>
        </>
      ) : (
        <>
          <img src={secondImg} style={{ width: "220px" }} />
          <div
            style={{
              fontSize: "48px",
              color: "#ff1493",
              fontWeight: "bold",
              marginTop: "20px",
            }}
          >
            Yay!!! ðŸ’–ðŸŽ‰
          </div>

          {/* Falling Hearts */}
          {hearts.map((heart) => (
            <div
              key={heart.id}
              style={{
                position: "fixed",
                top: "-10px",
                left: `${heart.left}%`,
                fontSize: "24px",
                animation: "fall 4s linear forwards",
              }}
            >
              ðŸ’—
            </div>
          ))}
        </>
      )}
    </div>
  );
}

function client() {
  createRoot(document.getElementById("root")).render(<App />);
}
if (typeof document !== "undefined") {
  client();
}
